FUNCTION_BLOCK COMM_MONITOR
VAR_INPUT
    COMM_TYPE : COMM_TYPE_ENUM;
    ALARM_TRIGGERED : BOOL;
    LOG_ENTRY : STRING;
VAR_OUTPUT
    IS_CONNECTED : BOOL;
    ERROR_OCCURRED : BOOL;
    ERROR_CODE : INT;
VAR
    LAST_CONNECTION_CHECK : TIME;
    CHECK_INTERVAL : TIME := T#1S; (* Adjust interval as needed *)
    AUDIT_TRAIL : ARRAY [0..100] OF STRUCT TIMESTAMP : TIME; STATUS : BOOL; ERROR : STRING; END_STRUCT;
    AUDIT_TRAIL_INDEX : INT := 0;
END_VAR

(* Poll Connection *)
IF T#1S > T#0S AND T#1S > LAST_CONNECTION_CHECK THEN
    LAST_CONNECTION_CHECK := T#1S;
    CASE COMM_TYPE OF
        OPC_UA:
            IS_CONNECTED := CHECK_OPC_UA_CONNECTION();
            IF NOT IS_CONNECTED THEN ERROR_OCCURRED := TRUE; ERROR_CODE := GET_OPC_UA_ERROR_CODE(); END_IF
        MODBUS:
            IS_CONNECTED := CHECK_MODBUS_CONNECTION();
            IF NOT IS_CONNECTED THEN ERROR_OCCURRED := TRUE; ERROR_CODE := GET_MODBUS_ERROR_CODE(); END_IF
        PROFINET:
            IS_CONNECTED := CHECK_PROFINET_CONNECTION();
            IF NOT IS_CONNECTED THEN ERROR_OCCURRED := TRUE; ERROR_CODE := GET_PROFINET_ERROR_CODE(); END_IF
    END_CASE
    
    (* Log Status *)
    AUDIT_TRAIL[AUDIT_TRAIL_INDEX].TIMESTAMP := T#1S;
    AUDIT_TRAIL[AUDIT_TRAIL_INDEX].STATUS := IS_CONNECTED;
    AUDIT_TRAIL[AUDIT_TRAIL_INDEX].ERROR := LOG_ENTRY;
    AUDIT_TRAIL_INDEX := (AUDIT_TRAIL_INDEX + 1) MOD 101; (* Circular buffer *)
END_IF

(* Trigger Alarm *)
IF ERROR_OCCURRED AND NOT ALARM_TRIGGERED THEN
    ALARM_TRIGGERED := TRUE;
    (* Send alarm notification *)
END_IF
END_FUNCTION_BLOCK
